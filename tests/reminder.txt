	  \v	   \m	    \M	     \V		matches ~
                'magic' 'nomagic'
	  $	       $	    $	     \$		matches end-of-line
	  .	       .	    \.	     \.		matches any character
	  *	       *	    \*	     \*		any number of the previous atom
	  ()	   \(\)     \(\)     \(\)	grouping into an atom
	  |	       \|	    \|	     \|		separating alternatives
	  \a	   \a	    \a	     \a		alphabetic character
	  \\	   \\	    \\	     \\		literal backslash
	  \.	   \.	    .	     .		literal dot
	  \{	   {	    {	     {		literal '{'
	  a	       a	    a	     a		literal 'a'
